- name: To install apache in two different target machine
  hosts: group
  become: yes
  tasks:
    - name: Install apache web server
      yum:
        name: httpd
        state: present
      notify: reload server

    - name: Editing configuration file using template in two different server
      template:
        src: apache.j2
        dest: /etc/httpd/conf/httpd.conf
      notify: reload server
 
  handlers:
    - name: reload server
      service:
         name: httpd
         state: restarted
